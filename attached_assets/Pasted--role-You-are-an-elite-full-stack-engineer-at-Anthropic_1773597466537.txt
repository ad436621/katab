<prisma_schema>
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider  = "postgresql"
  url       = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}

model Letter {
  id            String    @id @default(cuid())
  title         String
  body          String
  recipientName String
  uniqueToken   String    @unique
  isRead        Boolean   @default(false)
  readAt        DateTime?
  language      Language  @default(arabic)
  status        Status    @default(draft)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  questions     Question[]
  replies       Reply[]
  @@index([uniqueToken])
  @@map("letters")
}

model Question {
  id           String  @id @default(cuid())
  letterId     String
  questionText String
  answerText   String
  orderIndex   Int
  letter       Letter  @relation(fields: [letterId], references: [id], onDelete: Cascade)
  @@index([letterId, orderIndex])
  @@map("questions")
}

model Reply {
  id        String   @id @default(cuid())
  letterId  String
  replyBody String
  replyFrom String
  createdAt DateTime @default(now())
  letter    Letter   @relation(fields: [letterId], references: [id], onDelete: Cascade)
  @@map("replies")
}

model AdminSession {
  id           String   @id @default(cuid())
  sessionToken String   @unique
  expiresAt    DateTime
  createdAt    DateTime @default(now())
  @@index([sessionToken])
  @@map("admin_sessions")
}

enum Language { arabic english }
enum Status   { draft sent read replied }
</prisma_schema>

<auth_implementation>
Session system:

1. Login: POST /api/auth/login
   - Check username, bcrypt verify password, compare security answers
   - On success: crypto.randomUUID() token, store in admin_sessions with expiresAt = now+2h
   - Return { token, expiresAt }
   - On failure: 500ms delay then 401

2. Middleware (src/middleware/auth.js):
   - Extract Bearer token from Authorization header
   - Query admin_sessions: sessionToken matches AND expiresAt > now
   - If invalid: 401 JSON
   - If valid: extend expiresAt by 2 hours (sliding window), attach session to req
   - Export as: module.exports = requireAuth

3. Frontend (public/js/auth.js):
   const Auth = {
     SESSION_KEY: 'rls_session',
     getSession() { try { return JSON.parse(sessionStorage.getItem(this.SESSION_KEY)) } catch { return null } },
     setSession(data) { sessionStorage.setItem(this.SESSION_KEY, JSON.stringify(data)) },
     clearSession() { sessionStorage.removeItem(this.SESSION_KEY) },
     getToken() { return this.getSession()?.token ?? null },
     isValid() {
       const s = this.getSession()
       if (!s) return false
       return new Date(s.expiresAt) > new Date()
     },
     guardPage() { if (!this.isValid()) { window.location.replace('/login.html') } },
     startExpiryWatcher() {
       setInterval(() => {
         const s = this.getSession()
         if (!s) return
         const minsLeft = (new Date(s.expiresAt) - Date.now()) / 60000
         if (minsLeft <= 5 && minsLeft > 0) Utils.showExpiryWarning(Math.ceil(minsLeft))
         if (minsLeft <= 0) { this.clearSession(); window.location.replace('/login.html') }
       }, 60000)
     }
   }
</auth_implementation>
